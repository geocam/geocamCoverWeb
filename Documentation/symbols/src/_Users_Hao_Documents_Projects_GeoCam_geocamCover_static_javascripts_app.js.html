1 /** A Place consists of a name, an association with a marker,
  2   * as well as an array of reports and an array of tasks.
  3   * It also includes a position which is latitude and longitude.
  4   *
  5   * @class Creates a new Place
  6   * @property {Number} id The id of the Place.
  7   * @property {String} name The name of the Place.
  8   * @property {google.maps.LatLng} position The position object (from Google Maps) of the Place.
  9   * @property {Array} reports An array of Report objects associated with the Place.
 10   * @property {Array} tasks An array of Task objects associated with the Place.
 11   * @property {google.maps.Marker} marker The marker object of the Place.
 12   * @property {String} category The category of the Place.
 13   */
 14 function Place() {
 15     this.id = null;
 16     this.name = "";
 17     this.position = null;
 18     this.reports = [];
 19     this.tasks = [];
 20     this.marker = null;
 21     this.category = null;
 22 }

 24 /** The LogItem class is a superclass of Tasks and Reports.
 25   * They each share the members in this class.
 26   * The LogItem represents an entry inside the log for a Place.
 27   *
 28   * @class Creates a new LogItem
 29   * @property {Date} modified_at The last modified time of the LogItem.
 30   * @property {Number} id The id of the LogItem.
 31   * @property {Number} place_id The ID of the Place this LogItem is associated with.
 32   * @property {String} title The title given to the LogItem.
 33   */
 34 function LogItem() {
 35     this.modified_at = null;
 36     this.id = null;
 37     this.place_id = null;
 38     this.title = "";
 39 }

 41 /** The Task class inherits from the LogItem class.
 42   * The Task object represents an task to perform for the associated Place.
 43   *
 44   * @class Creates a new Task
 45   * @augments LogItem
 46   *
 47   * @property {String} description A description of the Task to perform for the Place.
 48   * @property {Number} priority The priority of performing this Task represented as a number from 0 to 2.
 49   * @property {Boolean} completed True or False representation of whether the Task is completed.
 50   */
 51 function Task() {
 52     this.description = "";
 53     this.priority = 0;
 54     this.completed = false;
 55 }
 56 Task.prototype = new LogItem();

 68   * @property {Number} task_id The optionally referenced Task's ID for this report.

 70 function Report() {
 71     this.task = null;
 72     this.status = "";
 73     this.percentCompleted = 0;
 74     this.notes = "";
 75     this.task_id = null;
 76 }

534 /** Determines which icon to draw for a place.
535   * Loops through all reports or tasks depending
536   * on which view is enabled. Currently, it renders the highest
537   * priority task icon or the most significant
538   * report status icon.
539   *
540   * @function
541   * @param {Place} place The Place object to add an icon to the map for.
542   */
543 function placeIcon(place) {
544     var icon;
545     if (selectedView == requestView) {
546         var priorityToDisplay = 0;
547         for (var t_id in place.tasks) {
548             if (!place.tasks[t_id].completed && place.tasks[t_id].priority > priorityToDisplay)
549                 priorityToDisplay = place.tasks[t_id].priority;
550         }
551         if (priorityToDisplay == 0)
552             icon = "whiteBox";
553         else
554             icon = "priority" + priorityToDisplay;
555     } else if (selectedView == reportView) {
556         var reportToDisplay = null;
557         var mostRecent = new Date();
558         mostRecent.setYear(1900);
559         for (var r_id in place.reports) {
560             if (place.reports[r_id].modified_at > mostRecent) {
561                 mostRecent = place.reports[r_id].modified_at;
562                 reportToDisplay = place.reports[r_id];
563             }
564         }
565         if (reportToDisplay) {
566             switch (parseInt(reportToDisplay.status)) {
567                 case 0:
568                 case 1:
569                     icon = "statusRed";
570                     break;
571                 case 2:
572                 case 3:
573                 case 4:
574                     icon = "statusYellow";
575                     break;
576                 case 5:
577                 case 6:
578                     icon = "statusGreen";
579                     break;
580                 default:
581                     icon = "whiteBox";
582             }
583         } else {
584             icon = "whiteBox";
585         }
586     }
587     return icon;
588 }

612 /** Grabs a place object from the place array.
613   * Renders the log view with the tasks and reports that the place has.
614   *
615   * @function
616   * @param {Number} place_id The Place object's ID to load the log screen for.
617   */
618 function showLog(place_id) {
619     place = places[place_id];
620     selectedPlace = place;
621     var noTasksAndReports = true;
622     var logList = new Array();
623     $('#place-name-h1').html(place.name.length == 0 ? 'Unnamed Place' : place.name);
624     $('#logs').empty();
625 
626     for (var t in place.tasks)
627         logList.push(place.tasks[t]);
628 
629     for (var r in place.reports)
630         logList.push(place.reports[r]);
631 
632     logList.sort(function(a, b) {
633         return b.modified_at - a.modified_at
634     });
635 
636     for (var log_id in logList) {
637         if (logList[log_id] instanceof Task) {
638             task = logList[log_id];
639             $('#logs').append("<li onclick='showEditTask(" + task.id + ");'> Task: " + task.description + " (" + task.modified_at_str + ")</li>");
640         } else {
641             report = logList[log_id];
642             $('#logs').append("<li onclick='showEditReport(" + report.id + ");'> Report: " + report.title + " (" + report.modified_at_str + ")</li>");
643         }
644         noTasksAndReports = false;
645     }
646 
647     if (noTasksAndReports)
648         $('#logs').append('<li>No tasks or reports</li>');
649 
650     try {
651         $('#logs').listview("refresh");
652     } catch(e) {
653     }
654     showPage("#logs-page");
655 }

846 /** Initiates the Geolocation module.
847   *
848   * @function
849   */
850 function initiateGeolocation() {
851     navigator.geolocation.getCurrentPosition(handleGeolocationQuery, handleErrors);
852 }

854 /** Handles the errors for the Geolocation module.
855   *
856   * @function
857   * @param {Object} error The error coming back from the Geolocation callback.
858   */
859 function handleErrors(error) {
860     gpsDenied = true;
861     switch (error.code) {
862         case error.PERMISSION_DENIED:
863             alert("user did not share geolocation data");
864             break;
865         case error.POSITION_UNAVAILABLE:
866             alert("could not detect current position");
867             break;
868         case error.TIMEOUT:
869             alert("retrieving position timed out");
870             break;
871         default:
872             alert("unknown error");
873             break;
874     }
875 }

893 
894   if (myMarker) {

899                 'title': "You are here",
900                 'icon': "/static/images/youAreHere.png",
901         'zIndex' : 1000
902             }, function(map, marker) {
903                 myMarker = marker;
904     });
905 
906     $(myMarker).click(function() {
907         clickedPosition = myLocation;
908         showPage('#place-form');
909     });
910 
911     $('#map_canvas').gmap({

913         });
914   }
